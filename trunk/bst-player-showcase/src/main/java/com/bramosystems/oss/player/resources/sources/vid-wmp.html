Widget c = null;
try {
    FlatVideoPlayer cap = new FlatVideoPlayer(Plugin.WinMediaPlayer,
        "http://bst-player.googlecode.com/svn/tags/showcase/media/teaching-of-islam.wmv",
        false, "350px", "100%");
    cap.showLogger(true);
    c = cap;
} catch (LoadException ex) {
    Window.alert("Load exp");
} catch (PluginVersionException ex) {
    c = PlayerUtil.getMissingPluginNotice(ex.getPlugin());
} catch (PluginNotFoundException ex) {
    c = PlayerUtil.getMissingPluginNotice(ex.getPlugin());
}
panel.add(c);
