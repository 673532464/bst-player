Playback with QuickTime Player plugin
Star Trek Movie Trailer (online)
Widget v = null;
try {
    v = new FlatVideoPlayer(
            Plugin.QuickTimePlayer,
            "http://movies.apple.com/" +
            "movies/paramount/star_trek" +
            "/startrek-tlr3_h.640.mov",
            false, "350px", "100%");
} catch (LoadException ex) {
    Window.alert("Load exp");
} catch (PluginVersionException ex) {
    v = PlayerUtil.getMissingPluginNotice(
            Plugin.QuickTimePlayer,
            ex.getRequiredVersion());
} catch (PluginNotFoundException ex) {
    v = PlayerUtil.getMissingPluginNotice(
            Plugin.QuickTimePlayer);
}
panel.add(v);
