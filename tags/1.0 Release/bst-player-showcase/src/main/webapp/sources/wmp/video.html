Widget wmp = null;
try {
   WinMediaPlayer p = new WinMediaPlayer(
        "http://bst-player.googlecode.com/" +
        "svn/tags/showcase/media/" +
        "islamic-jihad.wmv", false,
        "350px", "100%");
    p.showLogger(true);
    wmp = p;
} catch (LoadException ex) {
    Window.alert("Load exp");
} catch (PluginVersionException ex) {
    wmp = PlayerUtil.getMissingPluginNotice(
            Plugin.WinMediaPlayer);
} catch (PluginNotFoundException ex) {
    wmp = PlayerUtil.getMissingPluginNotice(
            Plugin.WinMediaPlayer);
}
panel.add(wmp);
