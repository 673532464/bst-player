    Capsule cap = new Capsule( GWT.getHostPageBaseURL() +

    cap.showLogger(true);

            "could not be found", false);

    c = PlayerUtil.getMissingPluginNotice(Plugin.Auto,
            "No Plugin", "A compatible sound plugin " +
            "could not be found", false);
}
