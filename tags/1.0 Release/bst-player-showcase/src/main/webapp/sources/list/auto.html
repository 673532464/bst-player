Widget v = null;
try {
    Capsule cc = new Capsule(Plugin.PlaylistSupport,
        GWT.getHostPageBaseURL() + "media/applause.mp3",
        false);
    cc.addToPlaylist(GWT.getHostPageBaseURL() +
        "media/o-na-som.mp3");
    cc.addToPlaylist(GWT.getHostPageBaseURL() +
        "media/thunder.mp3");
    cc.setShuffleEnabled(true);
    v = cc;
} catch (LoadException ex) {
    Window.alert("Load exp");
} catch (PluginVersionException ex) {
    v = PlayerUtil.getMissingPluginNotice(
            Plugin.PlaylistSupport, ex.getRequiredVersion());
} catch (PluginNotFoundException ex) {
    v = PlayerUtil.getMissingPluginNotice(
            Plugin.PlaylistSupport);
}
panel.add(v);
