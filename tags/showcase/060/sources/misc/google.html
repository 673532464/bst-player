Embedding Google Video
Bill Clinton at the Nigerian National Assembly (online)
Widget mp = null;
try {
    SWFWidget swf = new SWFWidget(
     "http://video.google.com/googleplayer.swf?" +
     "docid=-8228484564915781123&hl=en&fs=true",
     "100%", "350px", PluginVersion.get(9, 0, 0));
    swf.addProperty("bgcolor", "#000000");
    mp = swf;
} catch (PluginVersionException ex) {
    mp = SWFWidget.getMissingPluginNotice(
            PluginVersion.get(9, 0, 0));
} catch (PluginNotFoundException ex) {
    mp = SWFWidget.getMissingPluginNotice(
            PluginVersion.get(9, 0, 0));
}
panel.add(mp);
