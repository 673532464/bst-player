Widget w = null;
try {
    AbstractMediaPlayer mp = PlayerUtil.getPlayer(GWT.getHostPageBaseURL() +  "media/o-na-som.mp3",
            false, "50px", "100%");
    mp.setLoopCount(2);
    mp.showLogger(true);
    w = mp;
} catch (LoadException ex) {
    Window.alert("Load Exp");
} catch (PluginVersionException ex) {
    w = PlayerUtil.getMissingPluginNotice(Plugin.Auto);
} catch (PluginNotFoundException ex) {
    w = PlayerUtil.getMissingPluginNotice(Plugin.Auto);
}
panel.add(w);
